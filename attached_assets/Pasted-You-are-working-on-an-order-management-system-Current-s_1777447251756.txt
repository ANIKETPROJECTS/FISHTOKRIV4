### 3. Output Formatting (VERY IMPORTANT)

Before sending response:

* Convert all ObjectIds to string

Option A (recommended):
Use `.toJSON()` transform in schema:

```js
schema.set("toJSON", {
  transform: (doc, ret) => {
    ret._id = ret._id.toString();

    if (ret.customerId) ret.customerId = ret.customerId.toString();
    if (ret.subHubId) ret.subHubId = ret.subHubId.toString();
    if (ret.superHubId) ret.superHubId = ret.superHubId.toString();
    if (ret.couponId) ret.couponId = ret.couponId.toString();
    if (ret.timeslotId) ret.timeslotId = ret.timeslotId.toString();

    if (ret.couponIds) {
      ret.couponIds = ret.couponIds.map(id => id.toString());
    }

    if (ret.coupons) {
      ret.coupons = ret.coupons.map(c => ({
        ...c,
        id: c.id ? c.id.toString() : null
      }));
    }

    return ret;
  }
});
```
